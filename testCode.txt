@Test
    public void testGetHealthCheckSummaryDetails() {
        // Arrange
        GetSummaryDetailsRequest summaryDetailsRequest = new GetSummaryDetailsRequest();
        summaryDetailsRequest.setPageNum(1);
        summaryDetailsRequest.setPageSize(10);
        summaryDetailsRequest.setSortBy("execId");
        summaryDetailsRequest.setSortDir("asc");
        summaryDetailsRequest.setFilterCriteria(new ArrayList<>());

        HealthCheckSummary healthCheckSummary = new HealthCheckSummary();
        healthCheckSummary.setBusCheckCode("DQGdemo_data6");
        healthCheckSummary.setBusCheckGroup("DQGdemo_data6");

        Page<HealthCheckSummary> pagedHealthCheckSummaryList = new PageImpl<>(Collections.singletonList(healthCheckSummary));
        when(healthCheckSummaryRepository.findAll(any(Specification.class), any(Pageable.class))).thenReturn(pagedHealthCheckSummaryList);

        SummaryDetailsDTO summaryDetailsDTO = new SummaryDetailsDTO();
        summaryDetailsDTO.setGrouped(true);
        summaryDetailsDTO.setBusinessCheckGroup("DQGdemo_data6");
        summaryDetailsDTO.setTotalRecords(48);
        summaryDetailsDTO.setRecordsPassed(24);
        summaryDetailsDTO.setRecordsFailed(24);

        SummaryDetailsDTO summaryDetailsDT01 = new SummaryDetailsDTO();
        summaryDetailsDT01.setGrouped(true);
        summaryDetailsDT01.setBusinessCheckGroup("DQGdemo_data6");
        summaryDetailsDT01.setTotalRecords(48);
        summaryDetailsDT01.setRecordsPassed(24);
        summaryDetailsDT01.setRecordsFailed(24);

        List<SummaryDetailsDTO> dtos = new ArrayList<>();
        dtos.add(summaryDetailsDTO);
        dtos.add(summaryDetailsDT01);
        when(healthCheckSummaryMapper.toDtos(anyList())).thenReturn(dtos);

        // Act
        APIResponse<List<SummaryDetailsDTO>> apiResponse = processLogServiceImpl.getHealthCheckSummaryDetails(summaryDetailsRequest);

        // Assert
        assertThat(apiResponse).isNotNull();
        assertThat(apiResponse.getData()).isNotNull();
        assertThat(apiResponse.getData()).hasSize(2);
        assertThat(apiResponse.getData().get(0).getTotalRecords()).isEqualTo(48);
        assertThat(apiResponse.getData().get(0).getRecordsPassed()).isEqualTo(24);
        assertThat(apiResponse.getData().get(0).getRecordsFailed()).isEqualTo(24);
        assertThat(apiResponse.getData().get(1).getTotalRecords()).isEqualTo(48);
        assertThat(apiResponse.getData().get(1).getRecordsPassed()).isEqualTo(24);
        assertThat(apiResponse.getData().get(1).getRecordsFailed()).isEqualTo(24);
        assertThat(apiResponse.getMessage()).isEqualTo("Success");
    }
}
